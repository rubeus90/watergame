001package pkg_Room;
002import java.util.ArrayList;
003
004/**
005 * Cette classe gere la TransporterRoom, une salle qui, hormis mode teste, teleporte le joueur dans une salle aleatoire
006 * sur la carte
007 * 
008 * @author NGUYEN Hong Ngoc
009 * @author PATOIS Thibault
010 *
011 */
012public class TransporterRoom extends Room
013{
014        private RoomRandomizer random;
015        private boolean testMode;
016        private Room testModeRoom;
017        
018        /**
019         * Constructeur qui construit le TransporterRoom, qui herite da la classe Room, et qui est definie
020         * en plus par une ArrayList des salles du jeu
021         * 
022         * @param description
023         *                      La description de la salle
024         * @param image
025         *                      L'image de la salle
026         * @param nom
027         *                      Le nom de la salle
028         * @param rooms
029         *                      L'ArrayList des salles du jeu
030         */
031        public TransporterRoom(final String description, final String image, final String nom, final ArrayList<Room> rooms)
032        {
033                super(description, image, nom);
034                random = new RoomRandomizer(rooms);
035                testMode = false;
036        }
037        
038        /**
039         * Redefinir la methode getExit() de la classe Room. 
040         * 
041         * Ici, la methode permet en mode teste (avec la commande "aleas + nom de la salle) de retourner toujours la salle 
042         * correspondant au nom demande. Hors mode teste, elle retourne une salle aleatoire parmis les salles du jeu
043         */
044        @Override
045        public Room getExit(String direction)
046        {
047                if(testMode == false)
048                {
049                        return random.getRandomRoom();
050                }
051                else 
052                        return testModeRoom;
053                
054        }
055        
056        /**
057         * Imposer le mode teste
058         * 
059         * @param mode
060         *                      True si on veut activer le mode teste, false si on veut le desactiver
061         */
062        public void setTestMode(boolean mode)
063        {
064                testMode = mode;
065        }
066        
067        /**
068         * Imposer la salle qui sera retournee dans le mode teste
069         * 
070         * @param room
071         *                      La salle qui est retournee dans le mode teste
072         */
073        public void setTestModeRoom(Room room)
074        {
075                testModeRoom = room;
076        }
077}
