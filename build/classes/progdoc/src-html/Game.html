001import javax.swing.JApplet;
002import javax.swing.JOptionPane;
003
004import pkg_Characters.Player;
005import pkg_Game.GameEngine;
006import pkg_Game.UserInterface;
007
008/**
009 * Classe qui contient le main et qui génère tout le jeu
010 * 
011 * @author NGUYEN Hong Ngoc
012 * @author PATOIS Thibault
013 */
014public class Game extends JApplet 
015{
016        private static final long serialVersionUID = 1L;
017        
018        private static GameEngine engine;
019        private static UserInterface gui;
020        private static Player player;
021
022        /**
023         * Créer le jeu et initialiser les attributs
024         */
025        public Game()
026        {
027                String nom = JOptionPane.showInputDialog(null, "Entre ton nom:", "Nom du personnage", JOptionPane.QUESTION_MESSAGE);            
028                String[] gender = {"masculin", "féminin", "indéterminé"};
029                String sexe = (String)JOptionPane.showInputDialog(null, "Choisis ton sexe:", "Le sexe du personnage", JOptionPane.QUESTION_MESSAGE,
030              null,
031              gender,
032              gender[1]);       
033                
034                player = new Player(nom, sexe, 80);
035                engine = new GameEngine(player);
036                gui = new UserInterface(engine);
037                player.setGameEngine(engine);
038                player.setUserInterface(gui);
039                engine.setGUI(gui);
040        }
041        
042        /**
043         * Méthode main du jeu
044         */
045        public static void main(String[] args) 
046        {
047                new Game();
048        }
049        
050        /**
051         * Permet d'initialiser l'applet de jeu
052         */
053        @Override 
054        public void init()
055    {
056        super.init();
057    }
058}
