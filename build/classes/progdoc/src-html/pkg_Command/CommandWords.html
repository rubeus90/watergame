001package pkg_Command;
002import java.util.HashMap;
003import java.util.Set;
004
005
006/**
007 * Cette class enume toutes les commandes que le jeux connait. Elle permet de
008 * reconnaitre les commandes quand elles sont saisies.
009 * 
010 * @author NGUYEN Hong Ngoc
011 * @author PATOIS Thibault
012 * 
013 */
014
015public class CommandWords 
016{
017        private HashMap<CommandWord, String> validCommands;
018        private HashMap<String, Command> commands;
019
020        /**
021         * Instancier 2 HashMap, la premiere qui relie le String des commandes et les commandes correspondantes.
022         * La deuxieme HashMap relie le String de la commande a la classe de commmande associee
023         */
024        public CommandWords() 
025        {
026                validCommands = new HashMap<CommandWord, String>();
027
028                for (CommandWord command : CommandWord.values())
029                        validCommands.put(command, command.toString());
030                
031                commands = new HashMap<String, Command>();
032        commands.put(validCommands.get(CommandWord.GO), new GoCommand());
033        commands.put(validCommands.get(CommandWord.HELP), new HelpCommand());
034        commands.put(validCommands.get(CommandWord.QUIT), new QuitCommand());
035        commands.put(validCommands.get(CommandWord.LOOK), new LookCommand());
036        commands.put(validCommands.get(CommandWord.BACK), new BackCommand());
037        commands.put(validCommands.get(CommandWord.TEST), new TestCommand());
038        commands.put(validCommands.get(CommandWord.TAKE), new TakeCommand());
039        commands.put(validCommands.get(CommandWord.DROP), new DropCommand());
040        commands.put(validCommands.get(CommandWord.ITEMS), new ItemsCommand());
041        commands.put(validCommands.get(CommandWord.DRINK), new DrinkCommand());
042        commands.put(validCommands.get(CommandWord.UNKNOWN), new UnknownCommand());
043        commands.put(validCommands.get(CommandWord.TELEPORTER), new TeleporterCommand());
044        commands.put(validCommands.get(CommandWord.CHARGER), new ChargerCommand());
045        commands.put(validCommands.get(CommandWord.UTILISER), new UtiliserCommand());
046        commands.put(validCommands.get(CommandWord.ATTAQUE), new AttaqueCommand());
047        commands.put(validCommands.get(CommandWord.PARLER), new ParlerCommand());
048        commands.put(validCommands.get(CommandWord.ALEAS), new AleasCommand());
049        }
050        
051
052        /**
053         * Afficher tous les commandes
054         * 
055         * @return tous les commandes valables
056         */
057        public String getCommandlist() {
058                String Commandlist = "Les commandes sont:";
059                Set<CommandWord> keys = validCommands.keySet();
060                for (CommandWord commands : keys) {
061                        Commandlist += " " + commands.toString();
062                }
063                return Commandlist;
064        }
065
066        
067        /**
068         * Verifie si la commande rentree par l'utilisateur est une commande valide.
069         * 
070         * @return vraie si la commande saisie est une commande valide, faux si ce
071         *         n'est pas le cas.
072         */
073        public boolean isCommand(CommandWord aCommands) 
074        {
075                Set<CommandWord> keys = validCommands.keySet();
076                for (CommandWord commands : keys) {
077                        if (commands == aCommands)
078                                return true;
079                }
080                return false;
081        }
082
083        /**
084         * Retourner la commande correspondante au mot rentre par le joueur.
085         * Retourner une commande inconnue quand le mot rentre n'est pas une commande valable
086         * 
087         * @param word
088         * @return la commande correspondante avec le mot saisi
089         */
090        public Command get(String word)
091    {
092                if(commands.containsKey(word))
093                {
094                        return (Command)commands.get(word);
095                }
096                else
097                        return new UnknownCommand();
098    }
099}
