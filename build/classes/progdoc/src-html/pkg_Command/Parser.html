001package pkg_Command;
002import java.util.StringTokenizer;
003
004
005
006/**
007  * La class Parser ligne les entrees utilisateurs et essaye de les convertir en
008 * commandes pour le jeux. A chaque fois qu'elle est appelee, la class Parser
009 * lit les lignes du terminal et les decoupes en deux mots qui sont deux
010 * commandes. Cette class permet ensuite de convertir les commandes en un objet
011 * de la class Command.
012 * 
013 * La class Parser a un nombre determine de commande. La class Parser permet de
014 * comparer les entres du joueur avec les commandes connues et si la commande
015 * est inconnu, elle revoie un objet signe comme une command invalide.
016 * 
017 */
018public class Parser 
019{
020        private CommandWords commands;
021        
022        /**
023         * Constructeur qui permet de cree un parser qui va lire les entrees.
024         */
025        public Parser() 
026        {
027                commands = new CommandWords();
028//              reader = new Scanner(System.in);
029        }
030
031        /**
032         * Retourner la commande entree par l'utilisateur qui comporte 2 mots
033         * 
034         * @return la commande entree par l'utlisateur
035         */
036        public Command getCommand(String inputLine) 
037        {
038                String word = null;
039                String word2;
040
041                StringTokenizer tokenizer = new StringTokenizer(inputLine);
042
043                if (tokenizer.hasMoreTokens())
044                        word = tokenizer.nextToken(); //retourner le premier mot sous forme de String
045
046                if (tokenizer.hasMoreTokens())
047                        word2 = tokenizer.nextToken(); //retourner le second mot sous forme de String
048                else
049                        word2 = null; //si la commande entrée n'a pas de second mot
050
051                //On crée une commande à partir des 2 mots récupérés
052                Command command = commands.get(word);
053                command.setSecondWord(word2);
054                
055                return command;
056        }
057        
058        /**
059         * Retourner les commandes du jeu
060         * 
061         * @return les commandes du jeu
062         */
063        public String showCommands() {
064                return commands.getCommandlist();
065        }
066}
