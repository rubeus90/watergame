001package pkg_Command;
002import pkg_Characters.Player;
003
004/**
005 * Cette classe gere la commande "drink" du jeu, qui permet au joueur de boire une potion ou un soin
006 * 
007 * @author NGUYEN Hong Ngoc
008 * @author PATOIS Thibault
009 *
010 */
011public class DrinkCommand extends Command
012{
013        /**
014         * Constructeur qui permet de creer une commande Drink
015         */
016        public DrinkCommand()
017        {}
018        
019        /**
020         * Permettre au joueur de boire une potion ou un soin. Les potions et les soins ne sont pas des objets deplaceables
021         * mais ils sont fixes dans des salles. 
022         * Quand le joueur boire une potion ou un soin, la potion en question est retiree de la salle, et la sante
023         * du joueur augmente. L'augmentation depend de la potion bue (une potion ou un soin)
024         */
025        public void execute(Player player)
026        {
027                if (!this.hasSecondWord()) 
028                {
029                        player.getGUI().println("\n" + "Tu veux boire quoi crétin?");
030                } 
031                else 
032                {
033                        String mot = this.getSecondWord();
034
035                        //Si le joueur demande une potion
036                        if (mot.equals("potion")) 
037                        {
038                                if (player.getRoom().containPotion("Potion")) //s'il y a une potion dans la salle
039                                {
040                                        player.augmenteSante(10);
041                                        int index = player.getRoom().indexPotion("Potion");
042                                        player.getRoom().getArrayList().remove(index); //retirer la potion de la salle
043                                } 
044                                else //s'il n'y a pas de potion
045                                {
046                                        player.getGUI().println("\n" + "Il n'y a pas (ou plus) de cette potion dans cette salle");
047                                }
048
049                        }
050                        //Si le joueur demande un soin
051                        else if (mot.equals("soin")) 
052                        {
053                                if (player.getRoom().containPotion("Soin")) //s'il y a un soin dans la salle
054                                {
055                                        player.augmenteSante(20);
056                                        int index = player.getRoom().indexPotion("Soin");
057                                        player.getRoom().getArrayList().remove(index); //retirer le soin de la salle
058                                } 
059                                else //s'il n'y a pas de soin
060                                {
061                                        player.getGUI().println("Il n'y a pas (ou plus) de ce soin dans cette salle");
062                                }
063                        }
064
065                        else
066                                player.getGUI().println("\n" + "Ce que tu demandes de boire n'existe pas!");
067                }
068                        
069                        
070                        player.getGUI().println("\n" + "Ton niveau de santé est maintenant de " + player.getSante());
071        }
072        
073}
