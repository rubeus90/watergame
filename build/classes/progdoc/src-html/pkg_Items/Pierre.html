001package pkg_Items;
002
003/**
004 * Cette classe crée un téléporteur qui herite de la classe Item.
005 * 
006 * Le téléporteur est caractérisé par son état d'activation. Quand le joueur récupère 
007 * le téléporteur, cet objet est désactivé. Il doit le charger en récupérant un objet dans une autre salle. A chaque 
008 * utilisation, il est déchargé.
009 * 
010 * @author NGUYEN Hong Ngoc
011 * @author PATOIS Thibault
012 *
013 */
014public class Pierre extends Item
015{
016        private boolean activated;
017        
018        /**
019         * Constructeur qui construit la pierre magique 
020         */
021        public Pierre()
022        {
023                super("La pierre EnderPearl", 0);
024                activated = false;
025        }
026        
027        /**
028         * Retourner si oui ou non la pierre est chargee
029         * 
030         * @return true si la pierre est chargee, false sinon
031         */
032        public boolean getValueActivation()
033        {
034                return activated;
035        }
036        
037        /**
038         * Imposer l'etat d'activation de la pierre magique
039         * 
040         * @param etat
041         *                      Etat d'activation de la pierre (elle est chargee ou non)
042         */
043        public void setActivation(boolean etat)
044        {
045                activated = etat;
046        }
047
048}
