097   /**Ajouter un objet dans l'inventaire du joueur. L'objet pris est défini par le 2ème mot de la commande
098    * Par exemple: take arc
099    * 
100    * @param command
101    */
102   public void take(Command command)
103   {
104        String mot = command.getSecondWord();
105        Item item = aCurrentRoom.getItemListe().getValue(mot);
106        int poidsFuture = this.getPoidsInventaire() + item.getWeightItem();
107        
108        if(poidsFuture > maxPoids)
109                gui.println("Ton sac est déjà trop lourd, tu ne peux pas prendre plus d'objet. Jette un autre objet sinon!");
110        else
111        {        
112            if(!aCurrentRoom.getItemListe().containsKey(mot))
113                gui.println("Mais il n'y a pas de " + mot +" ici");
114            else
115            {
116//                Item item = aCurrentRoom.getItemListe().getValue(mot);           
117                items.putItem(mot, item);
118                aCurrentRoom.deleteItem(mot);
119                gui.println("Tu as pris un(e) " + mot);
120            }              
121        }
122        
123            
124   }

126   /**Retirer un objet de l'inventaire du joueur. L'objet retiré est défini par le 2ème mot de la commande
127    * Par exemple: drop arc
128    * 
129    * @param command
130    */
131   public void drop(Command command)
132   {
133        String mot = command.getSecondWord();   
134                   
135        if(!items.containsKey(mot))
136            gui.println("Tu n'as pas de " + mot);
137        else
138        {
139            Item item = items.getValue(mot);
140            items.removeItem(mot);
141            aCurrentRoom.addItem(mot, item);
142        }   
143   }

146   public String getInventaire()
147   {
148       String inventaire = "Dans ton inventaire: " + "\n";
149   
150       if(!items.getHashMap().isEmpty())
151       {
152           Set<String> keys = items.getKeys();
153           for(String nom : keys)
154           {
155               Item item = items.getValue(nom);
156               inventaire += item.getDescriptionItem() + " qui pèse " + item.getWeightItem() + "\n";
157           }
158       }
159       else
160           inventaire = "Il n'y a rien dans ton inventaire.";
161           
162       return inventaire;
163   }
164   
165   public int getPoidsInventaire()
166   {
167       Set<String> keys = items.getKeys();
168       int poids = 0;
169       for(String nom : keys)
170       {
171           Item item = items.getValue(nom);
172           poids += item.getWeightItem();
173       }
174       return poids;
175   }

183   /** 
184    * Procédure pour passer d'une salle à une autre. Si il n'y a pas de sortie, entré un nouvelle direction. 
185    * Sinon affiché un message d'erreur.
186    */
187   public void goRoom(Command command) 
188   {
189       if(!command.hasSecondWord()) 
190       {
191           gui.println("Aller ou exactement, soit un peu plus précis!");
192       }
193       
194       //Stocker la salle actuelle dans le stack (pour la méthode back)
195       salles.push(aCurrentRoom);
196       
197       String direction = command.getSecondWord();
198
199       Room nextRoom = aCurrentRoom.getExit(direction);
200
201       if (nextRoom == null) 
202       {
203           gui.println("Tu arrives au bord de l'île, tu vas tomber dans l'eau si tu continues!");
204       }
205       else 
206       {         
207           aCurrentRoom = nextRoom;
208//           player.setCurrentRoom(nextRoom);
209           engine.printLocationInfo();
210           
211           if(aCurrentRoom.getImageName() != null)
212           {
213               gui.showImage(aCurrentRoom.getImageName());
214           }
215       }
216   }

226   /**Implémenter la commande Back pour retourner dans la salle précédente.
227    * On stocke les salles visitée précédemment dans une Stack, à chaque fois on veut revenir sur notre pas, on utilise la commande "pop" qui prendre
228    * la denière valeur de la Stack (donc la dernière salle visitée) et qui retire cette salle de la Stack.
229    * Quand la Stack est vide, on est revenu au début du jeu.
230    */
231   public void back(Command command)
232   {
233       if(command.hasSecondWord())
234       {
235           gui.println("Tu ne peux pas revenir ou tu veux, c'est pas la fête ici!");
236       }
237       else
238       {
239           if(salles.empty() == true)
240               gui.println("Tu es revenu au début du jeu!");
241           else
242           {
243           aCurrentRoom = salles.pop();
244           engine.printLocationInfo();
245           
246           if(aCurrentRoom.getImageName() != null)
247               gui.showImage(aCurrentRoom.getImageName());
248               
249           }
250       }                 
251           
252   }

261   public void setMaxPoids()
262   {
263       if(sante>=0 && sante <20)
264            maxPoids = 0;
265       if(sante>=20 && sante < 60)
266            maxPoids = 70;
267       if(sante>=60 && sante < 80)
268            maxPoids = 100;
269       if(sante>=80 && sante <100)
270            maxPoids = 140;
271       if(sante >= 100)
272           maxPoids = 160;
273   }

308                   }

313           {
314                   if(aCurrentRoom.containPotion("Soin"))
315                   {
