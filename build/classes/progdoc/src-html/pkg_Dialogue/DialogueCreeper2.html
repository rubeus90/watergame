001package pkg_Dialogue;
002
003
004import pkg_Characters.Bots;
005import pkg_Game.GameEngine;
006import pkg_Items.Item;
007import pkg_Room.Room;
008
009public class DialogueCreeper2 extends Dialogue
010{
011        public DialogueCreeper2()
012        {
013                super();
014        }
015        
016        public void afficheDialogue(GameEngine engine)
017        {
018                if(engine.gameResetted())
019                {
020                        super.setEtape(1);
021                        engine.setResetGame();
022                }
023                
024                engine.getGUI().showDialogue(1);
025                                
026                if(engine.getPlayer().getItemListe().containsKey("papier"))
027                {
028                        if(super.getEtape() == 10)
029                        {
030                                super.setEtape(1);
031                        }
032                        
033                        switch(super.getEtape())
034                        {               
035                                case 1:
036                                {
037                                        engine.getGUI().println("Creeper : Alors tu revoilà! Je peux voir la feuille dans ta main?");
038                                        super.suivant();
039                                        break;
040                                }
041                                case 2:
042                                {
043                                        engine.getGUI().println("Creeper : RedStone? Cet inscrit me fait penser à un mythe de cette île. Les histoires disent qu'il" +
044                                                        "existe une salle secrète sur l'île, ou on peut trouver une hallebarde tellement puissante que l'on peut" +
045                                                        "vraincre n'importe quel ennemi!");
046                                        super.suivant();
047                                        break;
048                                }
049                                case 3:
050                                {
051                                        engine.getGUI().println("Creeper : Mais pour trouver cette salle secrète, il faut que l'on aille jusqu'au temple, " +
052                                                        "c'est là-bas que l'on trouvera le chemin");
053                                        super.suivant();
054                                        break;
055                                }
056                                case 4:
057                                {
058                                        engine.getGUI().println("Creeper : On ne peut pas y aller ensemble, c'est bien trop dangeureux. On se retrouve là-bas, " +
059                                                        "fais bien attention à toi!");
060                                        super.suivant();
061                                        break;
062                                }
063                                default: 
064                                {
065                                        engine.getPlayer().getRoom().removeBot("Creeper");
066                                        engine.getArrayListRoom().get(1).addBot("Creeper", new Bots("Creeper", null, 80, false));
067                                        engine.getArrayListRoom().get(1).setImage("images/temple.png");
068                                        engine.getPlayer().getRoom().setImage("images/grotte.png");
069                                        engine.getGUI().showImage(engine.getPlayer().getRoom().getImageName());
070                                        engine.getGUI().closeDialogue();                                
071                                }
072                        }
073                }
074                else
075                {
076                        if(engine.getArrayListRoom().get(6).containBot("Enderman"))
077                        {
078                                switch(super.getEtape())
079                                {
080                                        case 1:
081                                        {
082                                                engine.getGUI().println("Creeper : Tiens tu revoilà! Tu n'as pas croisé d'autres ennemis sur le chemin j'espère?");
083                                                super.suivant();
084                                                break;
085                                        }
086                                        case 2:
087                                        {
088                                                engine.getGUI().println("Creeper : Quoi? Tu as croisé Enderman mais tu l'as pas tué? C'est un ennemi puissante," +
089                                                                "c'est trop dangereux de le laisser en vie!");
090                                                super.suivant();
091                                                break;
092                                        }
093                                        case 3:
094                                        {
095                                                engine.getGUI().println("Creeper : De plus, l'épée que je t'ai donné est suffisamment puissante pour l'achever! Il faut" +
096                                                                "que tu y retourne impérativement pour finir la vie d'Enderman. Je t'attends ici!");
097                                                super.suivant();
098                                                break;
099                                        }
100                                        case 0:
101                                        {
102                                                engine.getGUI().println("Creeper : Tu attends quoi encore? Vas-y battre Enderman!");
103                                                super.setEtape(4);
104                                                break;
105                                        }
106                                        default: 
107                                        {
108                                                engine.getGUI().closeDialogue();                
109                                                super.setEtape(0);
110                                        }
111                                }
112                        }
113                        else
114                        {
115                                if(super.getEtape() == 0)
116                                {
117                                        super.setEtape(1);
118                                }
119                                switch(super.getEtape())
120                                {
121                                        case 1:
122                                        {
123                                                engine.getGUI().println("Creeper : Tiens tu revoilà! ");
124                                                super.suivant();
125                                                break;
126                                        }
127                                        case 2:
128                                        {
129                                                engine.getGUI().println("Creeper : Je vois que tu as réussi à tuer Enderman! C'est très bien!");
130                                                super.suivant();
131                                                break;
132                                        }
133                                        case 3:
134                                        {
135                                                engine.getGUI().println("Creeper : Mais j'ai entendu que Enderman a sur lui un papier qui contient une information" +
136                                                                "très importante, il doit être encore à l'endroit que tu as tué Enderman," +
137                                                                "vas le chercher, ça peut nous aider!");
138                                                super.suivant();
139                                                break;
140                                        }
141                                        case 10:
142                                        {
143                                                engine.getGUI().println("Creeper : Tu attends quoi encore? Vas chercher le papier!");
144                                                super.setEtape(4);
145                                                break;
146                                        }
147                                        default: 
148                                        {
149                                                engine.getGUI().closeDialogue();                
150                                                super.setEtape(10);
151                                        }
152                                }
153                        }
154                }
155        }
156        
157}
