001package pkg_Dialogue;
002
003import pkg_Characters.Bots;
004import pkg_Game.GameEngine;
005
006public class DialogueCreeper3 extends Dialogue
007{
008        public DialogueCreeper3()
009        {
010                super();
011        }
012        
013        public void afficheDialogue(GameEngine engine)
014        {
015                if(engine.gameResetted())
016                {
017                        super.setEtape(1);
018                        engine.setResetGame();
019                }
020                
021                engine.getGUI().showDialogue(1);
022                
023                if(engine.getPlayer().getItemListe().containsKey("papier"))
024                {
025                        if(super.getEtape() == 10)
026                        {
027                                super.setEtape(1);
028                        }
029                        switch(super.getEtape())
030                        {
031                                case 1:
032                                {               
033                                        engine.getGUI().println("Creeper : Donne moi voir ton papier. Dans ce temple il y a un flamme magique" +
034                                                        " qui nous permet de lire les écrits secrets qui sont invisibles à l'oeil nu, c'est peut-être " +
035                                                        "le cas de ce papier");
036                                        super.suivant();
037                                        break;
038                                }
039                                case 2:
040                                {
041                                        engine.getGUI().println("Creeper : J'avais bien raison. Ce papier contient le secret pour trouver la salle" +
042                                                        "secrète ou se trouve la hallebarde la plus puissante du monde qui te rendera invincible!");
043                                        super.suivant();
044                                        break;
045                                }
046                                default: 
047                                {
048                                        engine.getGUI().println("Creeper : Cette feuille t'as permis de trouver la porte pour aller dans la salle secrète! " +
049                                                        "Allons-y!");
050                                        engine.getGUI().createButtonDoor();
051                                        
052                                        engine.getPlayer().getRoom().removeBot("Creeper");
053                                        engine.getTransporterRoom().addBot("Creeper", new Bots("Creeper", null, 80, false));
054                                        engine.getPlayer().getRoom().setImage("images/temple.png");
055                                        engine.getTransporterRoom().setImage("images/SalleSecrete.png");
056                                }
057                        }
058                }
059                else
060                {
061                        switch(super.getEtape())
062                        {
063                                case 1:
064                                {
065                                        engine.getGUI().println("Creeper : Donne moi voir ton papier. Dans ce temple il y a un flamme magique" +
066                                                        " qui nous permet de lire les écrits secrets qui sont invisibles à l'oeil nu, c'est peut-être " +
067                                                        "le cas de ce papier");
068                                        super.suivant();
069                                        break;
070                                }
071                                case 2:
072                                {
073                                        engine.getGUI().println("Creeper : Mais alors tu as mis ou le papier? Il nous en faut pour pouvoir" +
074                                                        "trouver la porte vers la salle secrète!");
075                                        super.suivant();
076                                        break;
077                                }
078                                case 3:
079                                {
080                                        engine.getGUI().println("Creeper : Alors si tu l'as posé quelques parts, vas le chercher et reviens ici!");
081                                        super.suivant();
082                                        break;
083                                }
084                                case 10:
085                                {
086                                        engine.getGUI().println("Creeper : Tu attends quoi encore? Vas chercher le papier!");
087                                        super.setEtape(4);
088                                        break;
089                                }
090                                default: 
091                                {
092                                        engine.getGUI().closeDialogue();                
093                                        super.setEtape(10);
094                                }
095                        }
096                }
097                                
098        }
099}
