001package pkg_Room;
002import java.util.ArrayList;
003
004/**Cette classe gere la transporter room
005 * 
006 * @author rubeus
007 *
008 */
009public class TransporterRoom extends Room
010{
011        private RoomRandomizer random;
012        private boolean testMode;
013        private Room testModeRoom;
014        
015        public TransporterRoom(final String description, final String image, final String nom, final ArrayList<Room> rooms)
016        {
017                super(description, image, nom);
018                random = new RoomRandomizer(rooms);
019                testMode = false;
020        }
021        
022        @Override
023        public Room getExit(String direction)
024        {
025                if(testMode == false)
026                {
027                        return random.getRandomRoom();
028                }
029                else 
030                        return testModeRoom;
031                
032        }
033        
034        public void setTestMode(boolean mode)
035        {
036                testMode = mode;
037        }
038        
039        public void setTestModeRoom(Room room)
040        {
041                testModeRoom = room;
042        }
043}
