001package pkg_Command;
002import java.io.File;
003import java.io.FileNotFoundException;
004import java.util.Scanner;
005
006import pkg_Characters.Player;
007
008/**
009 * Cette classe gere la commande "test" du jeu, qui permet a partir d'un fichier de texte de lancer les commandes
010 * du jeu dans le but de tester le jeu plus rapidement
011 * 
012 * @author NGUYEN Hong Ngoc
013 * @author PATOIS Thibault
014 *
015 */
016public class TestCommand extends Command
017{
018        private Scanner sr;
019        
020        /**
021         * Constructeur qui permet de creer une commande Test
022         */
023        public TestCommand()
024        {}
025        
026        /**
027         * Permettre de lire les lignes d'un fichier de texte et d'executer ligne par ligne les commandes inscrites
028         */
029        public void execute(Player player)
030        {
031                try 
032                {
033                        if(this.hasSecondWord())
034                        {
035                                String secondWord = this.getSecondWord();
036                                
037                                //On lance les fichiers de teste différents en fonction du second mot de la commande
038                                switch(secondWord)
039                                {
040                                        case "gagner":
041                                        {
042                                                sr = new Scanner(new File("./src/Tests/gagner.txt"));
043                                                break;
044                                        }
045                                        case "commands":
046                                        {
047                                                sr = new Scanner(new File("./src/Tests/commands.txt"));
048                                                break;
049                                        }
050                                        case "perdre":
051                                        {
052                                                sr = new Scanner(new File("./src/Tests/perdre.txt"));
053                                                break;
054                                        }
055                                        case "salles":
056                                        {
057                                                sr = new Scanner(new File("./src/Tests/salles.txt"));
058                                                break;
059                                        }
060                                        default:
061                                                sr = null;
062                                                                        
063                                }                       
064                                if(sr != null)
065                                {
066                                        while (sr.hasNextLine()) //tant que le fichier de texte n'est pas fini, on lit et exécuter la ligne suivante
067                                        {
068                                                player.getGameEngine().interpretCommand(sr.nextLine());
069                                        }
070                                }
071                                else
072                                        player.getGUI().println("Je sais pas quoi tester....");
073                        }
074                        else
075                                player.getGUI().println("Je sais pas quoi tester....");
076                }
077                                                
078
079                catch (FileNotFoundException ex) 
080                {
081                        player.getGUI().println("Fichier non trouvé");
082                }
083                
084        }
085}
