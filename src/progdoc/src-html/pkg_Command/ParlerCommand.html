001package pkg_Command;
002
003import pkg_Characters.Player;
004import pkg_Dialogue.*;
005
006/**
007 * Cette classe gere la commande "parler" du jeu, qui permet au joueur d'interagir avec un personnage non joueur
008 * 
009 * @author NGUYEN Hong Ngoc
010 * @author PATOIS Thibault
011 *
012 */
013public class ParlerCommand extends Command
014{
015        private DialogueCreeper1 dialogueCreeper1;
016        private DialogueCreeper2 dialogueCreeper2;
017        private DialogueCreeper3 dialogueCreeper3;
018        private DialogueCreeper4 dialogueCreeper4;
019        private DialogueEnderman dialogueEnderman;
020        private DialogueBlaze dialogueBlaze;
021        
022        /**
023         * Constructeur qui permet de creer une commande Parler
024         * Instancier les differents dialogues avecs les ennemis
025         */
026        public ParlerCommand()
027        {
028                dialogueCreeper1 = new DialogueCreeper1();
029                dialogueCreeper2 = new DialogueCreeper2();
030                dialogueCreeper3 = new DialogueCreeper3();
031                dialogueCreeper4 = new DialogueCreeper4();
032                dialogueEnderman = new DialogueEnderman();
033                dialogueBlaze = new DialogueBlaze();
034        }
035        
036        /**
037         * Affiche des dialogues avec les ennemis dans le jeu
038         */
039        public void execute(Player player)
040        {
041                if(player.getRoom().getBot() != null)
042                {
043                        String nomBot = player.getRoom().getBot().getNom();
044                        String nomRoom = player.getRoom().getNomRoom();
045                        
046                        //Les dialogues sont différents pour chaque ennemi
047                        switch(nomBot)
048                        {
049                                case("Creeper"):
050                                {
051                                        //Le dialogue affiché dépend de la salle ou l'intéraction se passe
052                                        switch(nomRoom)
053                                        {
054                                                case "foret":
055                                                {
056                                                        dialogueCreeper1.afficheDialogue(player.getGameEngine());
057                                                        break;
058                                                }
059                                                case "grotte":
060                                                {
061                                                        dialogueCreeper2.afficheDialogue(player.getGameEngine());
062                                                        break;
063                                                }
064                                                case "temple":
065                                                {
066                                                        dialogueCreeper3.afficheDialogue(player.getGameEngine());
067                                                        break;
068                                                }
069                                                case "secret":
070                                                {
071                                                        dialogueCreeper4.afficheDialogue(player.getGameEngine());
072                                                        break;
073                                                }
074                                                default:{}
075                                        }
076                                        break;
077                                }
078                                case("Blaze"):
079                                {
080                                        dialogueBlaze.afficheDialogue(player.getGameEngine());
081                                        break;
082                                }
083                                case("Enderman"):
084                                {
085                                        dialogueEnderman.afficheDialogue(player.getGameEngine());
086                                        break;
087                                }
088                                default: {}
089                        }
090                }
091                else
092                        player.getGUI().println("Il n'y pas d'ennemis ici pour parler avec toi!");
093                
094                
095        }
096}
