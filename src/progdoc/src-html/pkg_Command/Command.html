001package pkg_Command;
002import pkg_Characters.Player;
003
004/**
005 * Cette classe gere les commandes du jeu rentrees par le joueur. C'est une classe abstraite qui permet d'implementer
006 * les classes de differentes commandes qui l'heritent.
007 * 
008 * Une commande est compose de deux mots : un mot qui definit la commande et le deuxieme qui precise un parametre de la commande.
009 * 
010 * Dans cette classe, on ne s'interesse qu'au deuxieme mot de la commande.
011 * Si la commande a seulement une mot, alors le second mot est null.
012 * 
013 * @author NGUYEN Hong Ngoc
014 * @author PATOIS Thibault
015 * 
016 */
017
018public abstract class Command 
019{
020        private String secondWord;
021
022        /**
023         * Instancie la commande. Par defaut, la commande n'a pas de deuxieme mot.
024         */
025        public Command()
026        {
027                secondWord = null;
028        }
029
030        /**
031         * Retourne le deuxieme mot de la commande. Si la commande n'a pas ete
032         * reconnu, le resultat sera NULL.
033         * 
034         * @return Le deuxi√®me mot de la commande.
035         */
036        public String getSecondWord() 
037        {
038                return secondWord;
039        }
040
041        /**
042         * Fonction boolean qui retourne vrai si la commande possede un second mot.
043         * 
044         * @return vrai si la commande possede un second mot
045         */
046        public boolean hasSecondWord() 
047        {
048                return (secondWord != null);
049        }
050
051        /**
052         * Imposer le second mot de la commande
053         * 
054         * @param secondWord
055         *                      Le deuxieme mot de la commande
056         */
057        public void setSecondWord(String secondWord)
058    {
059        this.secondWord = secondWord;
060    }
061        
062        /**
063         * Methode abstraite qui sera implementee dans les sous classes, qui permet d'executer les commandes
064         * 
065         * @param player
066         *                      Le joueur du jeu
067         */
068        public abstract void execute(Player player);
069}
