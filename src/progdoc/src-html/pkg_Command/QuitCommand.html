001package pkg_Command;
002import pkg_Characters.Player;
003
004/**
005 * Cette classe gere la commande "quit" du jeu, qui permet à l'utilisateur de quiter le jeu.
006 * 
007 * @author NGUYEN Hong Ngoc
008 * @author PATOIS Thibault
009 *
010 */
011public class QuitCommand extends Command
012{
013        /**
014         * Constructeur qui permet de creer une commande Quit
015         */
016        public QuitCommand()
017        {}
018        
019        /**
020         * Terminer le jeu et fermer la fenetre de jeu
021         */
022        public void execute(Player player)
023        {
024                if (this.hasSecondWord())
025                        player.getGUI().println("Quit quoi? Si tu veux quitter le jeux, écris juste quit");
026                else
027                        player.getGameEngine().endGame();
028        }
029}
