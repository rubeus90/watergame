075    /**
076     * Print out some text into the text area.
077     */
078    public void print(String text)
079    {
080        log.append(text);
081        log.setCaretPosition(log.getDocument().getLength());
082    }
083
084    /**
085     * Print out some text into the text area, followed by a line break.
086     */
087    public void println(String text)
088    {
089        log.append(text + "\n");
090        log.setCaretPosition(log.getDocument().getLength());
091    }

093    /**
094     * Show an image file in the interface.
095     */
096    public void showImage(String imageName)
097    {
098        URL imageURL = this.getClass().getClassLoader().getResource(imageName);
099        if(imageURL == null)
100            System.out.println("image not found");
101        else {
102            ImageIcon icon = new ImageIcon(imageURL);
103            image.setIcon(icon);
104            myFrame.pack();
105        }
106    }
107
108    /**
109     * Enable or disable input in the input field.
110     */
111    public void enable(boolean on)
112    {
113        entryField.setEditable(on);
114        if(!on)
115            entryField.getCaret().setBlinkRate(0);
116    }

294     * Actionlistener interface for entry textfield.

296    public void actionPerformed(ActionEvent e) 
297    {
298//        if(e.getSource() == bouton1) 
299//        {
300//            engine.interpretCommand("go");
301//        }
302        
303        if(e.getSource() == bouton2) 
304        {
305            engine.interpretCommand(new Command(CommandWord.GO, "nord"));
306        }       
307        else if(e.getSource() == bouton3) 
308        {
309            engine.interpretCommand(new Command(CommandWord.GO, "monter"));
310        }
311        else if(e.getSource() == bouton4) 
312        {
313            engine.interpretCommand(new Command(CommandWord.GO, "ouest"));
314        }
315//         else if(e.getSource() == bouton5) 
316//         {
317//          engine.interpretCommand();
318//         }
319        else if(e.getSource() == bouton6) 
320        {
321            engine.interpretCommand(new Command(CommandWord.GO, "est"));
322        }
323        else if(e.getSource() == bouton7) 
324        {
325            engine.interpretCommand(new Command(CommandWord.BACK, null));
326        }
327        else if(e.getSource() == bouton8) 
328        {
329            engine.interpretCommand(new Command(CommandWord.GO, "sud"));
330        }
331//         else if(e.getSource() == bouton9) 
332//         {
333//          engine.interpretCommand();
334//         }
335        else if(e.getSource() == bouton10) 
336        {
337            engine.interpretCommand(new Command(CommandWord.DRINK, "potion"));
338        }
339        else if(e.getSource() == bouton11) 
340        {
341            engine.interpretCommand(new Command(CommandWord.DRINK, "soin"));
342        }
343        else if(e.getSource() == bouton12) 
344        {
345            engine.interpretCommand(new Command(CommandWord.LOOK, null));
346        }
347        else if(e.getSource() == bouton13) 
348        {
349            engine.printInventaire();
350        }
351        else if(e.getSource() == authorAction)
352        {
353                engine.interpretCommand(new Command(CommandWord.CREDITS, null));
354        }
355        
356        else
357            processCommand();
358    }
359
360    /**
361     * A command has been entered. Read the command and do whatever is 
362     * necessary to process it.
363     */
364    private void processCommand()
365    {
366        String input = entryField.getText();
367        entryField.setText("");
368        Command command = parser.getCommand(input);
369
370        engine.interpretCommand(command);
371    }
