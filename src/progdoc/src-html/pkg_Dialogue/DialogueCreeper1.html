001package pkg_Dialogue;
002
003import pkg_Characters.Bots;
004import pkg_Game.GameEngine;
005import pkg_Items.Item;
006
007/**
008 * Cette classe gere le premier dialogue avec Creeper dans la foret
009 * 
010 * @author NGUYEN Hong Ngoc
011 * @author PATOIS Thibault
012 *
013 */
014public class DialogueCreeper1 extends Dialogue
015{
016        /**
017         * Constructeur pour le premier dialogue avec Creeper
018         */
019        public DialogueCreeper1()
020        {
021                super();
022        }
023        
024        /**
025         * Methode qui permet d'afficher le dialogue avec Creeper dans la foret
026         */
027        public void afficheDialogue(GameEngine engine)
028        {
029                if(engine.gameResetted()) //si le jeu est recommance, on remet l'etape a 1
030                {
031                        super.setEtape(1);
032                        engine.setResetGame(); //on remet le jeu comme s'il n'est pas recommence, sinon le dialogue va tourjous rester a l'etape 1
033                }
034                
035                engine.getGUI().showDialogue(1);
036                
037                if(!engine.getPlayer().getItemListe().containsKey("corde")) //si le joueur n'a pas de corde (logiquement le 1er dialogue)
038                {
039                        switch(super.getEtape())
040                        {
041                                case 1:
042                                {
043                                        engine.getGUI().showDialogue(2);
044                                        
045                                        engine.getGUI().println("Creeper : Hey hey, toi ! Oui toi là !  Je suis bloqué dans un trou juste là. Viens m'aider !");
046                                        super.suivant();
047                                        break;
048                                }
049                                case 2:
050                                {
051                                        engine.getGUI().println("Creeper : Quoi tu n'as pas de corde ? Tu comptais m'aider comment ? " +
052                                                                "Tu es élastigirl peut-être ? Je plaisante, t'inquiètes pas. " +
053                                                                "Je pense que tu pourra trouver une corde en haut du PIC. Si tu m'aides à sortir de là, " +
054                                                                "je te donnerai quelques conseils pour survivre sur cette île");
055                                        
056                                        super.suivant();
057                                        break;
058                                }
059                                case 3:
060                                {
061                                        engine.getGUI().println("Creeper : Mais il parraît que le chemin pour aller jusqu'au pic était détruit par une chute de pierre." +
062                                                        "Mon petit doigt me dit qu'une certaine pierre magique peut t'aider à aller n'importe ou tu veux sur l'île," +
063                                                        "c'est peut-être ça ce qu'il te faut....");
064                                        super.suivant();
065                                        break;
066                                }
067                                case 0:
068                                {
069                                        engine.getGUI().println("Creeper : Tu attends quoi encore? Va chercher une corde au pic pour m'aider");
070                                        super.setEtape(4);
071                                        break;
072                                }
073                                default: 
074                                {
075                                        engine.getGUI().closeDialogue();
076                                        super.setEtape(0);
077                                }
078                        }
079                }
080                else //si le joueur a une corde pour l'aider (logiquement le 2eme dialogue)
081                {
082                        if(super.getEtape() == 0)
083                        {
084                                super.setEtape(1);
085                        }
086                        
087                        switch(super.getEtape())
088                        {
089                                case 1:
090                                {
091                                        engine.getGUI().println("Creeper : Ah voilà tu es de retours! Je vois que tu as trouvé une corde." +
092                                                        "Maintenant utilise la corde pour me sortir de là!");
093                                        super.suivant();
094                                        break;
095                                }
096                                case 2:
097                                {
098                                        engine.getGUI().println("Creeper : Merci de m'avoir sauvé. Pour te remercier, je vais te donner cette épée.");
099                                        engine.getPlayer().getItemListe().putItem("epee", new Item("Une épée en acier rare", 2));
100                                        super.suivant();
101                                        break;
102                                }
103                                case 3:
104                                {
105                                        engine.getGUI().println("Creeper : Allons-nous en d'ici, c'est trop risqué de rester ici!");
106                                        super.suivant();
107                                        break;
108                                }
109                                case 4:
110                                {
111                                        engine.getGUI().println("Creeper : Il faut que l'on se sépare. Si tu veux me retrouver, je serai dans la " +
112                                                        "grotte pas loin d'ici.");
113                                        super.suivant();
114                                        break;
115                                }
116                                default: 
117                                {
118                                        engine.getPlayer().getRoom().removeBot("Creeper");
119                                        engine.getArrayListRoom().get(4).addBot("Creeper", new Bots("Creeper", null, 80, false));
120                                        engine.getArrayListRoom().get(4).setImage("images/grotteAvecCreeper.png");
121                                        engine.getPlayer().getRoom().setImage("images/foret.png");
122                                        engine.getGUI().showImage(engine.getPlayer().getRoom().getImageName());
123                                        engine.getGUI().closeDialogue();
124                                        
125                                }
126                        }
127                }
128                
129        }
130}
