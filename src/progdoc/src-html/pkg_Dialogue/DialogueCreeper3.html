001package pkg_Dialogue;
002
003import pkg_Characters.Bots;
004import pkg_Game.GameEngine;
005
006/**
007 * Cette classe gere le troisieme dialogue avec Creeper dans le temple
008 * 
009 * @author NGUYEN Hong Ngoc
010 * @author PATOIS Thibault
011 *
012 */
013public class DialogueCreeper3 extends Dialogue
014{
015        /**
016         * Constructeur pour le troisieme dialogue avec Creeper
017         */
018        public DialogueCreeper3()
019        {
020                super();
021        }
022        
023        /**
024         * Methode qui permet d'afficher le dialogue avec Creeper dans le temple
025         */
026        public void afficheDialogue(GameEngine engine)
027        {
028                if(engine.gameResetted()) //si le jeu est recommance, on remet l'etape a 1
029                {
030                        super.setEtape(1);
031                        engine.setResetGame(); //on remet le jeu comme s'il n'est pas recommence, sinon le dialogue va tourjous rester a l'etape 1
032                }
033                
034                engine.getGUI().showDialogue(1);
035                
036                if(engine.getPlayer().getItemListe().containsKey("papier")) //si le joueur a toujour le papier sur lui
037                {
038                        if(super.getEtape() == 10)
039                        {
040                                super.setEtape(1);
041                        }
042                        switch(super.getEtape())
043                        {
044                                case 1:
045                                {               
046                                        engine.getGUI().println("Creeper : Donne moi voir ton papier. Dans ce temple il y a un flamme magique" +
047                                                        " qui nous permet de lire les écrits secrets qui sont invisibles à l'oeil nu, c'est peut-être " +
048                                                        "le cas de ce papier");
049                                        super.suivant();
050                                        break;
051                                }
052                                case 2:
053                                {
054                                        engine.getGUI().println("Creeper : J'avais bien raison. Ce papier contient le secret pour trouver la salle" +
055                                                        "secrète ou se trouve la hallebarde la plus puissante du monde qui te rendera invincible!");
056                                        super.suivant();
057                                        break;
058                                }
059                                default: 
060                                {
061                                        engine.getGUI().println("Creeper : Cette feuille t'as permis de trouver la porte pour aller dans la salle secrète! " +
062                                                        "Allons-y!");
063                                        engine.getGUI().createButtonDoor();
064                                        
065                                        engine.getPlayer().getRoom().removeBot("Creeper");
066                                        engine.getTransporterRoom().addBot("Creeper", new Bots("Creeper", null, 80, false));
067                                        engine.getPlayer().getRoom().setImage("images/temple.png");
068                                        engine.getTransporterRoom().setImage("images/sallesecretecreeper.png");
069                                }
070                        }
071                }
072                else //si le joueur a jete le papier quelques parts, demande lui d'aller le chercher
073                {
074                        switch(super.getEtape())
075                        {
076                                case 1:
077                                {
078                                        engine.getGUI().println("Creeper : Donne moi voir ton papier. Dans ce temple il y a un flamme magique" +
079                                                        " qui nous permet de lire les écrits secrets qui sont invisibles à l'oeil nu, c'est peut-être " +
080                                                        "le cas de ce papier");
081                                        super.suivant();
082                                        break;
083                                }
084                                case 2:
085                                {
086                                        engine.getGUI().println("Creeper : Mais alors tu as mis ou le papier? Il nous en faut pour pouvoir" +
087                                                        "trouver la porte vers la salle secrète!");
088                                        super.suivant();
089                                        break;
090                                }
091                                case 3:
092                                {
093                                        engine.getGUI().println("Creeper : Alors si tu l'as posé quelques parts, vas le chercher et reviens ici!");
094                                        super.suivant();
095                                        break;
096                                }
097                                case 10:
098                                {
099                                        engine.getGUI().println("Creeper : Tu attends quoi encore? Vas chercher le papier!");
100                                        super.setEtape(4);
101                                        break;
102                                }
103                                default: 
104                                {
105                                        engine.getGUI().closeDialogue();                
106                                        super.setEtape(10);
107                                }
108                        }
109                }
110                                
111        }
112}
